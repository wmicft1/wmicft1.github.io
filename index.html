# 列表推导式示例
numbers = [1, 2, 3, 4, 5]
squares = [x**2 for x in numbers]
print("平方数:", squares)

# 字典操作示例
person = {"name": "张三", "age": 25, "city": "北京"}
print("个人信息:", person)

# 函数定义示例
def calculate_average(numbers):
    return sum(numbers) / len(numbers) if numbers else 0

# 使用示例
grades = [85, 90, 78, 92, 88]
average = calculate_average(grades)
print("平均分:", average)
